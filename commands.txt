PostgreSQL Commands
===================

# --- psql (Windows - adjust path if needed) ---
SET PSQL="C:\Program Files\PostgreSQL\18\bin\psql.exe"

# Connect
%PSQL% -h localhost -p 5433 -U postgres -d payguard_ai

# List databases
%PSQL% -h localhost -p 5433 -U postgres -l

# List tables
%PSQL% -h localhost -p 5433 -U postgres -d payguard_ai -c "\dt"

# Show all data
%PSQL% -h localhost -p 5433 -U postgres -d payguard_ai -c "SELECT id, full_name, email, business_name FROM users;"
%PSQL% -h localhost -p 5433 -U postgres -d payguard_ai -c "SELECT id, business_name, risk_category FROM customers;"
%PSQL% -h localhost -p 5433 -U postgres -d payguard_ai -c "SELECT id, invoice_number, status, total_amount, pending_amount FROM invoices;"
%PSQL% -h localhost -p 5433 -U postgres -d payguard_ai -c "SELECT id, invoice_id, amount, payment_mode FROM payments;"
%PSQL% -h localhost -p 5433 -U postgres -d payguard_ai -c "SELECT id, invoice_id, followup_type, status FROM followups;"
%PSQL% -h localhost -p 5433 -U postgres -d payguard_ai -c "SELECT user_id, ai_provider, currency FROM settings;"

# Count records
%PSQL% -h localhost -p 5433 -U postgres -d payguard_ai -c "SELECT 'users' as tbl, COUNT(*) FROM users UNION ALL SELECT 'customers', COUNT(*) FROM customers UNION ALL SELECT 'invoices', COUNT(*) FROM invoices UNION ALL SELECT 'payments', COUNT(*) FROM payments UNION ALL SELECT 'followups', COUNT(*) FROM followups UNION ALL SELECT 'settings', COUNT(*) FROM settings;"

# --- Drop and recreate database ---
%PSQL% -h localhost -p 5433 -U postgres -c "DROP DATABASE IF EXISTS payguard_ai;"
%PSQL% -h localhost -p 5433 -U postgres -c "CREATE DATABASE payguard_ai;"

# --- Start the backend ---
cd backend
python -m uvicorn server:app --host 0.0.0.0 --port 8000 --reload

# --- Test API ---
curl http://localhost:8000/api/
curl -X POST http://localhost:8000/api/auth/login -H "Content-Type: application/json" -d "{\"email\":\"demo@payguard.ai\",\"password\":\"demo123\"}"

# --- Check logs ---
Get-Content logs\app.log -Tail 10
Get-Content logs\access.log -Tail 10
Get-Content logs\error.log -Tail 10
